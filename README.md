operator=input()
integer_1=int(input())
integer_2=int(input())

if operator== "+":
    print(integer_1 + integer_2)
elif operator=="-":
    print(integer_1 - integer_2)
elif operator=="*":
    print(integer_1 * integer_2)
elif operator=="/":
    print(integer_1 / integer_2)
elif operator=="%":
    print(integer_1 % integer_2)
